final class SOS_PG_Plugin
{
    private static ?SOS_PG_Plugin $instance = null;

    public static function instance(): SOS_PG_Plugin
    {
        if (self::$instance === null) {
            self::$instance = new self();
        }

        return self::$instance;
    }

    private function __construct()
    {
        $this->load_files();
        $this->boot();
    }

    private function load_files(): void
    {
        require_once SOS_PG_DIR . 'includes/Core/Response.php';
        require_once SOS_PG_DIR . 'includes/Core/Logger.php';
        require_once SOS_PG_DIR . 'includes/Core/Auth.php';

        require_once SOS_PG_DIR . 'includes/Api/Partner_Routes.php';

        require_once SOS_PG_DIR . 'includes/WP/Settings.php';
        require_once SOS_PG_DIR . 'includes/WP/Hooks.php';
    }

    private function boot(): void
    {
        add_action('rest_api_init', ['SOS_PG_Partner_Routes', 'register']);
        add_action('admin_menu', ['SOS_PG_Settings', 'register_menu']);
        add_action('admin_init', ['SOS_PG_Settings', 'register_settings']);
        add_action('init', ['SOS_PG_Hooks', 'init']);
    }
}

class SOS_PG_Logger
{
    public static function enabled(): bool
    {
        return (bool) get_option('sos_pg_enable_logs', 1);
    }

    public static function log(string $event, array $context = []): void
    {
        if (!self::enabled()) {
            return;
        }

        $upload_dir = wp_upload_dir();
        $dir = trailingslashit($upload_dir['basedir']) . 'sos-partner-gateway';

        if (!file_exists($dir)) {
            wp_mkdir_p($dir);
        }

        $file = trailingslashit($dir) . 'gateway.log';

        $row = [
            'time'    => current_time('mysql'),
            'event'   => $event,
            'context' => $context,
            'ip'      => self::get_ip(),
            'ua'      => isset($_SERVER['HTTP_USER_AGENT']) ? sanitize_text_field(wp_unslash($_SERVER['HTTP_USER_AGENT'])) : '',
        ];

        error_log(wp_json_encode($row, JSON_UNESCAPED_UNICODE) . PHP_EOL, 3, $file);
    }

    private static function get_ip(): string
    {
        $keys = ['HTTP_CF_CONNECTING_IP', 'HTTP_X_FORWARDED_FOR', 'REMOTE_ADDR'];

        foreach ($keys as $key) {
            if (!empty($_SERVER[$key])) {
                $value = sanitize_text_field(wp_unslash($_SERVER[$key]));
                if ($key === 'HTTP_X_FORWARDED_FOR') {
                    $parts = explode(',', $value);
                    return trim($parts[0]);
                }
                return $value;
            }
        }

        return '';
    }
}

<?php

if (!defined('ABSPATH')) {
    exit;
}

class SOS_PG_Auth
{
    public static function get_partner_secret(): string
    {
        return (string) get_option('sos_pg_partner_secret', '');
    }

    public static function verify_shared_secret(?string $provided): bool
    {
        $expected = self::get_partner_secret();

        if ($expected === '' || $provided === null || $provided === '') {
            return false;
        }

        return hash_equals($expected, $provided);
    }

    public static function issue_token(array $payload = []): string
    {
        $data = [
            'iss' => home_url(),
            'iat' => time(),
            'exp' => time() + 3600,
            'sub' => $payload['partner_id'] ?? 'partner',
            'ctx' => $payload,
        ];

        $json = wp_json_encode($data);
        return base64_encode($json ?: '{}');
    }

    public static function validate_token(string $token): ?array
    {
        $decoded = base64_decode($token, true);
        if ($decoded === false) {
            return null;
        }

        $data = json_decode($decoded, true);
        if (!is_array($data)) {
            return null;
        }

        if (empty($data['exp']) || time() > (int) $data['exp']) {
            return null;
        }

        return $data;
    }
}

<?php

if (!defined('ABSPATH')) {
    exit;
}

class SOS_PG_Partner_Routes
{
    public static function register(): void
    {
        register_rest_route('sos/v1', '/health', [
            'methods'             => WP_REST_Server::READABLE,
            'callback'            => [self::class, 'health'],
            'permission_callback' => '__return_true',
        ]);

        register_rest_route('sos/v1', '/partner/login', [
            'methods'             => WP_REST_Server::CREATABLE,
            'callback'            => [self::class, 'partner_login'],
            'permission_callback' => '__return_true',
        ]);

        register_rest_route('sos/v1', '/partner/verify', [
            'methods'             => WP_REST_Server::READABLE,
            'callback'            => [self::class, 'partner_verify'],
            'permission_callback' => '__return_true',
        ]);

        register_rest_route('sos/v1', '/partner/log', [
            'methods'             => WP_REST_Server::CREATABLE,
            'callback'            => [self::class, 'partner_log'],
            'permission_callback' => '__return_true',
        ]);
    }

    public static function health(WP_REST_Request $request): WP_REST_Response
    {
        return SOS_PG_Response::success([
            'plugin'  => 'SOS Partner Gateway',
            'version' => SOS_PG_VERSION,
            'status'  => 'ok',
        ]);
    }

    public static function partner_login(WP_REST_Request $request): WP_REST_Response
    {
        $secret = $request->get_header('x-partner-secret');
        $body   = $request->get_json_params();

        if (!SOS_PG_Auth::verify_shared_secret($secret)) {
            SOS_PG_Logger::log('partner_login_failed', [
                'reason' => 'invalid_secret',
            ]);

            return SOS_PG_Response::error('Unauthorized', 401);
        }

        $partner_id   = isset($body['partner_id']) ? sanitize_text_field((string) $body['partner_id']) : 'partner';

        if (!file_exists($dir)) {

            'uri'     => isset($_SERVER['REQUEST_URI']) ? sanitize_text_field(wp_unslash($_SERVER['REQUEST_URI'])) : '',

    private static function get_ip(): string
    {
        $keys = ['HTTP_CF_CONNECTING_IP', 'HTTP_X_FORWARDED_FOR', 'REMOTE_ADDR'];

        foreach ($keys as $key) {
            if (!empty($_SERVER[$key])) {
                $value = sanitize_text_field(wp_unslash($_SERVER[$key]));
                if ($key === 'HTTP_X_FORWARDED_FOR') {
                    $parts = explode(',', $value);
                    return trim($parts[0]);
                }
                return $value;
            }
        }

        return '';
    }

class SOS_PG_Auth
{
    public static function get_partner_secret(): string
    {
        return (string) get_option('sos_pg_partner_secret', '');
    }

    public static function verify_shared_secret(?string $provided): bool
    {
        $expected = self::get_partner_secret();

        if ($expected === '' || $provided === null || $provided === '') {
            return false;
        }

        return hash_equals($expected, $provided);
    }

    public static function issue_token(array $payload = []): string
    {
        $secret = self::get_partner_secret();
        if ($secret === '') {
            $secret = wp_salt('auth');
        }

        $data = [
            'iss' => home_url(),
            'iat' => time(),
            'exp' => time() + 3600,
            'sub' => $payload['partner_id'] ?? 'partner',
            'ctx' => $payload,
        ];

        $json = wp_json_encode($data);
        if ($json === false) {
            $json = '{}';
        }

        $payload_encoded = self::base64url_encode($json);
        $signature = hash_hmac('sha256', $payload_encoded, $secret, true);
        $signature_encoded = self::base64url_encode($signature);

        return $payload_encoded . '.' . $signature_encoded;
    }

    public static function validate_token(string $token): ?array
    {
        $secret = self::get_partner_secret();
        if ($secret === '') {
            $secret = wp_salt('auth');
        }

        $parts = explode('.', $token);
        if (count($parts) !== 2) {
            return null;
        }

        [$payload_encoded, $signature_encoded] = $parts;

        $expected_signature = self::base64url_encode(
            hash_hmac('sha256', $payload_encoded, $secret, true)
        );

        if (!hash_equals($expected_signature, $signature_encoded)) {
            return null;
        }

        $decoded = self::base64url_decode($payload_encoded);
        if ($decoded === false) {
            return null;
        }

        $data = json_decode($decoded, true);
        if (!is_array($data)) {
            return null;
        }

        if (empty($data['exp']) || time() > (int) $data['exp']) {
            return null;
        }

        return $data;
    }

    public static function get_bearer_token(): string
    {
        $header = '';

        if (isset($_SERVER['HTTP_AUTHORIZATION'])) {
            $header = sanitize_text_field(wp_unslash($_SERVER['HTTP_AUTHORIZATION']));
        } elseif (isset($_SERVER['REDIRECT_HTTP_AUTHORIZATION'])) {
            $header = sanitize_text_field(wp_unslash($_SERVER['REDIRECT_HTTP_AUTHORIZATION']));
        }

        if (stripos($header, 'Bearer ') === 0) {
            return trim(substr($header, 7));
        }

        return '';
    }

    private static function base64url_encode(string $data): string
    {
        return rtrim(strtr(base64_encode($data), '+/', '-_'), '=');
    }

    private static function base64url_decode(string $data)
    {
        $remainder = strlen($data) % 4;
        if ($remainder) {
            $data .= str_repeat('=', 4 - $remainder);
        }

        return base64_decode(strtr($data, '-_', '+/'), true);
    }
}

    public static function capture_token_from_query(): void
    {
        if (is_admin()) {
            return;
        }

        if (empty($_GET['sos_pg_token'])) {
            return;
        }

        $token = sanitize_text_field(wp_unslash($_GET['sos_pg_token']));
        $data  = SOS_PG_Auth::validate_token($token);

        if ($data === null) {
            SOS_PG_Logger::log('partner_token_capture_failed', [
                'reason' => 'invalid_token',
            ]);
            return;
        }

        setcookie(
            'sos_pg_token',
            $token,
            [
                'expires'  => time() + 3600,
                'path'     => COOKIEPATH ? COOKIEPATH : '/',
                'secure'   => is_ssl(),
                'httponly' => true,
                'samesite' => 'Lax',
            ]
        );

        $_COOKIE['sos_pg_token'] = $token;

        SOS_PG_Logger::log('partner_token_captured', [
            'partner_id' => $data['ctx']['partner_id'] ?? '',
        ]);
    }

    public static function handle_partner_page_access(): void
    {
        if (is_admin() || !is_singular('page')) {
            return;
        }

        global $post;
        if (!$post || empty($post->ID)) {
            return;
        }

        $page_id = (int) $post->ID;
        $partners = SOS_PG_Partner_Config::get_all();

        if (empty($partners)) {
            return;
        }

        $matched_partner = null;

        foreach ($partners as $partner) {
            if (!is_array($partner)) {
                continue;
            }

            $protected_page_id = isset($partner['protected_page_id']) ? (int) $partner['protected_page_id'] : 0;
            if ($protected_page_id === $page_id) {
                $matched_partner = $partner;
                break;
            }
        }

        if (!$matched_partner) {
            return;
        }

        $token = '';

        if (!empty($_GET['sos_pg_token'])) {
            $token = sanitize_text_field(wp_unslash($_GET['sos_pg_token']));
        } elseif (!empty($_COOKIE['sos_pg_token'])) {
            $token = sanitize_text_field(wp_unslash($_COOKIE['sos_pg_token']));
        }

        if ($token === '') {
            SOS_PG_Logger::log('partner_page_blocked', [
                'reason'   => 'missing_token',
                'page_id'  => $page_id,
                'partner'  => $matched_partner['partner_id'] ?? '',
            ]);

            wp_die(
                esc_html__('Accesso non autorizzato: token mancante.', 'sos-partner-gateway'),
                esc_html__('Accesso negato', 'sos-partner-gateway'),
                ['response' => 403]
            );
        }

        $data = SOS_PG_Auth::validate_token($token);
        if ($data === null) {
            SOS_PG_Logger::log('partner_page_blocked', [
                'reason'   => 'invalid_token',
                'page_id'  => $page_id,
                'partner'  => $matched_partner['partner_id'] ?? '',
            ]);

            wp_die(
                esc_html__('Accesso non autorizzato: token non valido.', 'sos-partner-gateway'),
                esc_html__('Accesso negato', 'sos-partner-gateway'),
                ['response' => 403]
            );
        }

        $ctx = isset($data['ctx']) && is_array($data['ctx']) ? $data['ctx'] : [];
        $token_partner_id = isset($ctx['partner_id']) ? sanitize_text_field((string) $ctx['partner_id']) : '';
        $configured_partner_id = isset($matched_partner['partner_id']) ? sanitize_text_field((string) $matched_partner['partner_id']) : '';

        if ($token_partner_id !== $configured_partner_id) {
            SOS_PG_Logger::log('partner_page_blocked', [
                'reason'             => 'partner_mismatch',
                'page_id'            => $page_id,
                'token_partner_id'   => $token_partner_id,
                'expected_partner_id'=> $configured_partner_id,
            ]);

            wp_die(
                esc_html__('Accesso non autorizzato: partner non valido.', 'sos-partner-gateway'),
                esc_html__('Accesso negato', 'sos-partner-gateway'),
                ['response' => 403]
            );
        }

        SOS_PG_Logger::log('partner_page_access_granted', [
            'page_id'    => $page_id,
            'partner_id' => $token_partner_id,
        ]);
    }

        $keys = ['HTTP_CF_CONNECTING_IP', 'HTTP_X_FORWARDED_FOR', 'REMOTE_ADDR'];

        foreach ($keys as $key) {
            if (!empty($_SERVER[$key])) {
                $value = sanitize_text_field(wp_unslash($_SERVER[$key]));

                if ($key === 'HTTP_X_FORWARDED_FOR') {
                    $parts = explode(',', $value);
                    return trim($parts[0]);
                }

                return $value;
            }
        }

        return '';
    }

class SOS_PG_Auth
{
    public static function get_partner_secret(): string
    {
        return (string) get_option('sos_pg_partner_secret', '');
    }

    public static function verify_shared_secret(?string $provided): bool
    {
        $expected = self::get_partner_secret();

        if ($expected === '' || $provided === null || $provided === '') {
            return false;
        }

        return hash_equals($expected, $provided);
    }

    public static function issue_token(array $payload = [], int $ttl = 3600): string
    {
        $secret = self::get_signing_secret();

        $data = [
            'iss' => home_url(),
            'iat' => time(),
            'exp' => time() + $ttl,
            'sub' => $payload['partner_id'] ?? 'partner',
            'ctx' => $payload,
        ];

        $json = wp_json_encode($data);
        if ($json === false) {
            $json = '{}';
        }

        $payload_encoded = self::base64url_encode($json);
        $signature = hash_hmac('sha256', $payload_encoded, $secret, true);
        $signature_encoded = self::base64url_encode($signature);

        return $payload_encoded . '.' . $signature_encoded;
    }

    public static function validate_token(string $token): ?array
    {
        $secret = self::get_signing_secret();

        $parts = explode('.', $token);
        if (count($parts) !== 2) {
            return null;
        }

        [$payload_encoded, $signature_encoded] = $parts;

        $expected_signature = self::base64url_encode(
            hash_hmac('sha256', $payload_encoded, $secret, true)
        );

        if (!hash_equals($expected_signature, $signature_encoded)) {
            return null;
        }

        $decoded = self::base64url_decode($payload_encoded);
        if ($decoded === false) {
            return null;
        }

        $data = json_decode($decoded, true);
        if (!is_array($data)) {
            return null;
        }

        if (empty($data['exp']) || time() > (int) $data['exp']) {
            return null;
        }

        return $data;
    }

    public static function get_bearer_token(): string
    {
        $header = '';

        if (isset($_SERVER['HTTP_AUTHORIZATION'])) {
            $header = sanitize_text_field(wp_unslash($_SERVER['HTTP_AUTHORIZATION']));
        } elseif (isset($_SERVER['REDIRECT_HTTP_AUTHORIZATION'])) {
            $header = sanitize_text_field(wp_unslash($_SERVER['REDIRECT_HTTP_AUTHORIZATION']));
        }

        if (stripos($header, 'Bearer ') === 0) {
            return trim(substr($header, 7));
        }

        return '';
    }

    private static function get_signing_secret(): string
    {
        $secret = self::get_partner_secret();

        if ($secret === '') {
            $secret = wp_salt('auth');
        }

        return $secret;
    }

    private static function base64url_encode(string $data): string
    {
        return rtrim(strtr(base64_encode($data), '+/', '-_'), '=');
    }

    private static function base64url_decode(string $data)
    {
        $remainder = strlen($data) % 4;

        if ($remainder) {
            $data .= str_repeat('=', 4 - $remainder);
        }

        return base64_decode(strtr($data, '-_', '+/'), true);
    }
}

class SOS_PG_Security
{
    public static function get_request_ip(): string
    {
        return SOS_PG_Logger::get_ip();
    }

    public static function is_ip_allowed_for_partner(string $partner_id): bool
    {
        $allowed_ips = SOS_PG_Partner_Config::get_allowed_ips($partner_id);

        if (empty($allowed_ips)) {
            return true;
        }

        $request_ip = self::get_request_ip();
        if ($request_ip === '') {
            return false;
        }

        foreach ($allowed_ips as $allowed_ip) {
            if (self::ip_matches($request_ip, $allowed_ip)) {
                return true;
            }
        }

        return false;
    }

    private static function ip_matches(string $request_ip, string $allowed_ip): bool
    {
        $request_ip = trim($request_ip);
        $allowed_ip = trim($allowed_ip);

        if ($request_ip === '' || $allowed_ip === '') {
            return false;
        }

        if (strpos($allowed_ip, '/') === false) {
            return hash_equals($allowed_ip, $request_ip);
        }

        [$subnet, $mask] = explode('/', $allowed_ip, 2);
        $mask = (int) $mask;

        if (filter_var($request_ip, FILTER_VALIDATE_IP, FILTER_FLAG_IPV4) === false) {
            return false;
        }

        if (filter_var($subnet, FILTER_VALIDATE_IP, FILTER_FLAG_IPV4) === false) {
            return false;
        }

        if ($mask < 0 || $mask > 32) {
            return false;
        }

        $request_long = ip2long($request_ip);
        $subnet_long = ip2long($subnet);

        if ($request_long === false || $subnet_long === false) {
            return false;
        }

        $mask_long = -1 << (32 - $mask);

        return (($request_long & $mask_long) === ($subnet_long & $mask_long));

    public static function partner_login(WP_REST_Request $request): WP_REST_Response
    {
        $secret = $request->get_header('x-partner-secret');
        $body   = $request->get_json_params();

        if (!is_array($body)) {
            $body = [];
        }

        if (!SOS_PG_Auth::verify_shared_secret($secret)) {
            SOS_PG_Logger::log('partner_login_failed', [
                'reason' => 'invalid_secret',
            ]);

            return SOS_PG_Response::error('Unauthorized', 401);
        }

        $partner_id = isset($body['partner_id']) ? sanitize_text_field((string) $body['partner_id']) : '';
        if ($partner_id === '') {
            return SOS_PG_Response::error('partner_id mancante', 400);
        }

        if (!SOS_PG_Security::is_ip_allowed_for_partner($partner_id)) {
            SOS_PG_Logger::log('partner_login_blocked', [
                'reason'     => 'ip_not_allowed',
                'partner_id' => $partner_id,
            ]);

            return SOS_PG_Response::error('IP non autorizzato', 403);
        }

        $configured_redirect = SOS_PG_Partner_Config::get_redirect_url($partner_id);
        $body_redirect = isset($body['redirect_url']) ? esc_url_raw((string) $body['redirect_url']) : '';
        $redirect_url = $configured_redirect !== '' ? $configured_redirect : $body_redirect;

        $token = SOS_PG_Auth::issue_token([
            'partner_id'   => $partner_id,
            'redirect_url' => $redirect_url,
            'request_ip'   => SOS_PG_Security::get_request_ip(),
        ]);

        SOS_PG_Logger::log('partner_login_success', [
            'partner_id'   => $partner_id,
            'redirect_url' => $redirect_url,
        ]);

        return SOS_PG_Response::success([
            'token'        => $token,
            'partner_id'   => $partner_id,
            'redirect_url' => $redirect_url,
            'expires_in'   => 3600,
        ]);
    }

class SOS_PG_Hooks
{
    public static function init(): void
    {
        add_action('init', [self::class, 'capture_token_from_query'], 1);
        add_action('template_redirect', [self::class, 'handle_partner_page_access'], 1);
    }

    public static function capture_token_from_query(): void
    {
        if (is_admin()) {
            return;
        }

        if (empty($_GET['sos_pg_token'])) {
            return;
        }

        $token = sanitize_text_field(wp_unslash($_GET['sos_pg_token']));
        $data  = SOS_PG_Auth::validate_token($token);

        if ($data === null) {
            SOS_PG_Logger::log('partner_token_capture_failed', [
                'reason' => 'invalid_token',
            ]);
            return;
        }

        setcookie(
            'sos_pg_token',
            $token,
            [
                'expires'  => time() + 3600,
                'path'     => COOKIEPATH ? COOKIEPATH : '/',
                'secure'   => is_ssl(),
                'httponly' => true,
                'samesite' => 'Lax',
            ]
        );

        $_COOKIE['sos_pg_token'] = $token;

        SOS_PG_Logger::log('partner_token_captured', [
            'partner_id' => $data['ctx']['partner_id'] ?? '',
        ]);
    }

    public static function handle_partner_page_access(): void
    {
        if (is_admin() || !is_singular('page')) {
            return;
        }

        global $post;

        if (!$post || empty($post->ID)) {
            return;
        }

        $page_id = (int) $post->ID;
        $partners = SOS_PG_Partner_Config::get_all();

        if (empty($partners)) {
            return;
        }

        $matched_partner = null;

        foreach ($partners as $partner) {
            if (!is_array($partner)) {
                continue;
            }

            $protected_page_id = isset($partner['protected_page_id']) ? (int) $partner['protected_page_id'] : 0;

            if ($protected_page_id === $page_id) {
                $matched_partner = $partner;
                break;
            }
        }

        if (!$matched_partner) {
            return;
        }

        $token = '';

        if (!empty($_GET['sos_pg_token'])) {
            $token = sanitize_text_field(wp_unslash($_GET['sos_pg_token']));
        } elseif (!empty($_COOKIE['sos_pg_token'])) {
            $token = sanitize_text_field(wp_unslash($_COOKIE['sos_pg_token']));
        }

        if ($token === '') {
            SOS_PG_Logger::log('partner_page_blocked', [
                'reason'  => 'missing_token',
                'page_id' => $page_id,
                'partner' => $matched_partner['partner_id'] ?? '',
            ]);

            wp_die(
                esc_html__('Accesso non autorizzato: token mancante.', 'sos-partner-gateway'),
                esc_html__('Accesso negato', 'sos-partner-gateway'),
                ['response' => 403]
            );
        }

        $data = SOS_PG_Auth::validate_token($token);

        if ($data === null) {
            SOS_PG_Logger::log('partner_page_blocked', [
                'reason'  => 'invalid_token',
                'page_id' => $page_id,
                'partner' => $matched_partner['partner_id'] ?? '',
            ]);

            wp_die(
                esc_html__('Accesso non autorizzato: token non valido.', 'sos-partner-gateway'),
                esc_html__('Accesso negato', 'sos-partner-gateway'),
                ['response' => 403]
            );
        }

        $ctx = isset($data['ctx']) && is_array($data['ctx']) ? $data['ctx'] : [];

        $token_partner_id = isset($ctx['partner_id']) ? sanitize_text_field((string) $ctx['partner_id']) : '';
        $configured_partner_id = isset($matched_partner['partner_id']) ? sanitize_text_field((string) $matched_partner['partner_id']) : '';

        if ($token_partner_id !== $configured_partner_id) {
            SOS_PG_Logger::log('partner_page_blocked', [
                'reason'              => 'partner_mismatch',
                'page_id'             => $page_id,
                'token_partner_id'    => $token_partner_id,
                'expected_partner_id' => $configured_partner_id,
            ]);

            wp_die(
                esc_html__('Accesso non autorizzato: partner non valido.', 'sos-partner-gateway'),
                esc_html__('Accesso negato', 'sos-partner-gateway'),
                ['response' => 403]
            );
        }

        if (!SOS_PG_Security::is_ip_allowed_for_partner($configured_partner_id)) {
            SOS_PG_Logger::log('partner_page_blocked', [
                'reason'     => 'ip_not_allowed',
                'page_id'    => $page_id,
                'partner_id' => $configured_partner_id,
            ]);

            wp_die(
                esc_html__('Accesso non autorizzato: IP non consentito.', 'sos-partner-gateway'),
                esc_html__('Accesso negato', 'sos-partner-gateway'),
                ['response' => 403]
            );
        }

        SOS_PG_Logger::log('partner_page_access_granted', [
            'page_id'    => $page_id,
            'partner_id' => $configured_partner_id,
        ]);
    }
}

Questa base usa HMAC SHA-256 per il token, non ECC.
È una base buona per partire subito e collegare Angular. La struttura è già pronta per passare a ECC dopo senza rifare le API.

Non ho capito che significa non ECC.

Significa questo:

Adesso il token è firmato con una secret condivisa

cioè una password lunga uguale su entrambi i lati.

Nel codice che ti ho dato c’è questa logica:

WordPress conosce una secret

anche il partner conosce la stessa secret

con quella secret si firma/verifica il token

Questo si chiama HMAC, non ECC.

ECC invece cos’è

ECC = crittografia a chiave pubblica/privata

Funziona così:

il partner ha una chiave privata

WordPress ha solo la chiave pubblica

il partner firma

WordPress verifica

Quindi WordPress non conosce il segreto privato.

Differenza pratica
HMAC

C’è un solo segreto condiviso.

Partner <-> stessa secret <-> WordPress

Se quella secret viene rubata:

chiunque può firmare richieste valide

ECC

Ci sono 2 chiavi diverse.

Partner -> chiave privata
WordPress -> chiave pubblica

Se rubano la chiave pubblica:

non possono firmare nulla

Se rubano la privata:

sì, ma almeno il server non la conserva

Quindi perché ti ho dato HMAC ora?

Per partire più veloce.

Perché:

è più semplice da implementare

è più facile da testare

per integrare WordPress + Angular basta subito

non rompe la struttura

👉 In pratica è una versione iniziale funzionante

Esempio semplice
HMAC

È come se due persone avessero la stessa password segreta.

Per aprire una porta:

uno manda messaggio + firma con password

l’altro controlla usando la stessa password

ECC

È come se una persona avesse:

chiave privata per firmare

chiave pubblica da dare agli altri per controllare

Gli altri possono verificare, ma non possono imitarti.

class SOS_PG_ECC
{
    public static function verify_signature(string $payload, string $signature_base64, string $public_key): bool
    {
        if ($payload === '' || $signature_base64 === '' || $public_key === '') {
            return false;
        }

        $signature = base64_decode($signature_base64, true);

        if ($signature === false) {
            return false;
        }

        $result = openssl_verify(
            $payload,
            $signature,
            $public_key,
            OPENSSL_ALGO_SHA256
        );

        return $result === 1;
    }
}
